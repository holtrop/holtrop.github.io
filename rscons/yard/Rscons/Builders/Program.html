def new(options, *more)
  unless File.basename(options[:target])["."]
    options[:target] += options[:env].expand_varref("${PROGSUFFIX}", options[:vars])
  end
  if options[:vars][:direct]
    Object.new(options, *more)
  else
    super
  end
end
